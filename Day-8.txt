ğŸ“ Jenkins Pipeline with Docker (Python App)
pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                git 'https://github.com/your-repo/python-app.git'
            }
        }
        stage('Build Docker Image') {
            steps {
                script {
                    sh 'docker build -t python-app .'
                }
            }
        }
        stage('Run Container') {
            steps {
                script {
                    sh 'docker run -d -p 5000:5000 python-app'
                }
            }
        }
    }

    post {
        success {
            echo "Python app deployed successfully using Docker!"
        }
        failure {
            echo "Build failed. Please check logs."
        }
    }
}
